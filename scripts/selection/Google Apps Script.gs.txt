/**
 * Mixtape generator - 一键生成 Side A/B、时间码、干净歌单、画布占位
 * 说明：源表名必须为 "tracklist"，且第一行为标题，随后的每行：Name, Duration
 */

function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('🎛 Mixtape')
    .addItem('🎚 One-Click Full Generate', 'generateFullMixtape')
    .addToUi();
}

/** 解析各种可能的 duration 表示（数字秒 / mm:ss / hh:mm:ss / Date） -> 秒数 */
function parseDuration(durationStr) {
  if (durationStr === null || durationStr === undefined || durationStr === "") return 0;

  // 已经是数字（秒）
  if (typeof durationStr === 'number') {
    return Math.round(durationStr);
  }

  // 如果是 Date 对象（少见），取分秒
  if (Object.prototype.toString.call(durationStr) === "[object Date]") {
    return durationStr.getMinutes() * 60 + durationStr.getSeconds();
  }

  // 字符串处理
  const s = String(durationStr).trim();
  if (!s) return 0;

  // 如果是纯数字的字符串（秒）
  if (/^\d+$/.test(s)) return parseInt(s, 10);

  const parts = s.split(':').map(p => p.trim());
  if (parts.length === 2) {
    const m = parseInt(parts[0], 10) || 0;
    const sec = parseInt(parts[1], 10) || 0;
    return m * 60 + sec;
  } else if (parts.length === 3) {
    const h = parseInt(parts[0], 10) || 0;
    const m = parseInt(parts[1], 10) || 0;
    const sec = parseInt(parts[2], 10) || 0;
    return h * 3600 + m * 60 + sec;
  }
  return 0;
}

function formatSeconds(sec) {
  const m = Math.floor(sec / 60);
  const s = sec % 60;
  return `${m}:${s.toString().padStart(2,'0')}`;
}

function shuffleArray(arr) {
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
}

/** 主入口：生成 A/B、写入新 Sheet、生成时间码/干净歌单、插入画布占位 */
function generateFullMixtape() {
  const ui = SpreadsheetApp.getUi();
  try {
    Logger.log("🔔 generateFullMixtape 开始");
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    if (!ss) {
      ui.alert("找不到活跃的 Spreadsheet");
      return;
    }
    const trackSheet = ss.getSheetByName('tracklist');
    if (!trackSheet) {
      ui.alert("未找到名为 'tracklist' 的工作表，请确保存在并且第一行是标题");
      Logger.log("❌ tracklist 表不存在");
      return;
    }

    const raw = trackSheet.getDataRange().getValues();
    if (!raw || raw.length <= 1) {
      ui.alert("tracklist 中没有数据（或只有标题）。");
      Logger.log("❌ tracklist 数据为空");
      return;
    }

    // 取出数据（跳过标题）
    const rows = raw.slice(1).map(r => [ (r[0] || "").toString().trim(), r[1] ]);
    Logger.log(`✅ 读取 ${rows.length} 条原始记录`);

    // 如果源数据只有 name + seconds（秒为 number），允许；否则 parseDuration 会处理
    shuffleArray(rows);
    Logger.log("🔀 已打乱原始列表");

    const MIN_DURATION = 1800; // 30 min
    const MAX_DURATION = 1850; // small slack

    const sideA = [];
    const sideB = [];
    let totalA = 0, totalB = 0;

    for (const [name, durRaw] of rows) {
      const dur = parseDuration(durRaw);
      if (!name || !dur || isNaN(dur)) continue;

      // 优先填充 still-under-min side，防止一面空
      if (totalA < MIN_DURATION && (totalA + dur <= MAX_DURATION)) {
        sideA.push([name, dur]);
        totalA += dur;
      } else if (totalB < MIN_DURATION && (totalB + dur <= MAX_DURATION)) {
        sideB.push([name, dur]);
        totalB += dur;
      } else {
        // 当两面都达到 MIN 时停止
        if (totalA >= MIN_DURATION && totalB >= MIN_DURATION) break;
        // 若某一面已满但还没另一面，则尝试加入另一面（更随机）
        if (totalA < MIN_DURATION && (totalA + dur <= MAX_DURATION)) {
          sideA.push([name, dur]);
          totalA += dur;
        } else if (totalB < MIN_DURATION && (totalB + dur <= MAX_DURATION)) {
          sideB.push([name, dur]);
          totalB += dur;
        }
      }
    }

    // 如果未能满足 min（可能歌曲过短/数量不足）仍继续尽量填充
    for (const [name, durRaw] of rows) {
      const dur = parseDuration(durRaw);
      if (!name || !dur || isNaN(dur)) continue;
      // 跳过已在侧面里的
      if (sideA.some(x => x[0] === name) || sideB.some(x => x[0] === name)) continue;
      if (totalA < MIN_DURATION) { sideA.push([name,dur]); totalA += dur; }
      else if (totalB < MIN_DURATION) { sideB.push([name,dur]); totalB += dur; }
      if (totalA >= MIN_DURATION && totalB >= MIN_DURATION) break;
    }

    shuffleArray(sideA);
    shuffleArray(sideB);

    Logger.log(`🅰️ Side A: ${sideA.length} tracks, ${totalA}s`);
    Logger.log(`🅱️ Side B: ${sideB.length} tracks, ${totalB}s`);

    // 新建 sheet（包含时间）
    const now = new Date();
    const sheetName = Utilities.formatDate(now, Session.getScriptTimeZone(), "HH:mm:ss dd/MM/yyyy");
    const newSheet = ss.insertSheet(sheetName);
    if (!newSheet) throw new Error("无法创建新工作表");

    // 准备写入 A/B 列表（四列：Side, Track Name, Duration (sec), Duration (mm:ss)）
    const rowsToWrite = [];
    const pushFormatted = (label, list, total) => {
      for (const t of list) {
        rowsToWrite.push([label, t[0], t[1], formatSeconds(t[1])]);
      }
      rowsToWrite.push([label, '(Subtotal)', total, formatSeconds(total)]);
    };

    pushFormatted("A", sideA, totalA);
    rowsToWrite.push(["", "", "", ""]);
    pushFormatted("B", sideB, totalB);

    newSheet.getRange(1,1,rowsToWrite.length,4).setValues(rowsToWrite);
    newSheet.autoResizeColumns(1,4);
    Logger.log(`✅ 新 Sheet "${sheetName}" 写入完成，共 ${rowsToWrite.length} 行`);

    // 在新 sheet 上生成时间码和干净歌单并写入
    generateMixAndTrackListInSheet(newSheet);

    // 插入画布占位（chart）
    insertCanvasHeader(newSheet);

    ui.alert("✅ Mixtape 生成完成: " + sheetName);
    Logger.log("🔔 generateFullMixtape 完成");
  } catch (err) {
    Logger.log("❌ generateFullMixtape 出错: " + (err && err.message));
    SpreadsheetApp.getUi().alert("生成出错: " + (err && err.message));
    throw err;
  }
}

/** 在传入的 sheet 上生成带 Needle 的 FFmpeg 时间码 (G 列) 和无 Needle 的干净歌单 (I 列) */
function generateMixAndTrackListInSheet(sheet) {
  try {
    if (!sheet) throw new Error("generateMixAndTrackListInSheet: sheet 未定义");
    Logger.log("▶ generateMixAndTrackListInSheet 开始 for sheet: " + sheet.getName());

    const data = sheet.getDataRange().getValues();
    Logger.log(`读取 sheet 数据行数: ${data.length}, 列数: ${sheet.getMaxColumns()}`);

    const mixTimeline = [];
    const cleanTimeline = [];
    let currentTimeSec = 0;

    function toSecondsFromCell(v) {
      return parseDuration(v); // 假设你已有这个函数
    }

    function formatTime(seconds) {
      const totalSeconds = Math.floor(seconds);
      const minutes = Math.floor(totalSeconds / 60);
      const secs = totalSeconds % 60;
      return `${minutes}:${secs.toString().padStart(2, '0')}`;
    }

    function addSide(sideLabel) {
      Logger.log(`--- addSide ${sideLabel} start (time ${currentTimeSec}s)`);
      mixTimeline.push(`${formatTime(currentTimeSec)} Needle On Vinyl Record`);
      currentTimeSec += 3;

      for (let r = 1; r < data.length; r++) {
        const row = data[r];
        const side = row[0];
        const title = row[1];
        const durCell = row[2] || row[3];
        if (side !== sideLabel || !title || String(title).trim() === "(Subtotal)") continue;

        const dur = toSecondsFromCell(durCell);
        if (!dur || dur <= 0) {
          Logger.log(`跳过无效时长行: ${title} (${durCell})`);
          continue;
        }

        const timestamp = formatTime(currentTimeSec);
        mixTimeline.push(`${timestamp} ${title}`);
        cleanTimeline.push(`${timestamp} ${title}`);

        currentTimeSec += Math.max(0, dur - 2);
        mixTimeline.push(`${formatTime(currentTimeSec)} Needle Noise`);
        currentTimeSec += 5;
      }

      Logger.log(`--- addSide ${sideLabel} end (time ${currentTimeSec}s)`);
    }

    addSide("A");
    currentTimeSec += 3;
    addSide("B");

    Logger.log(`mixTimeline length=${mixTimeline.length}, cleanTimeline length=${cleanTimeline.length}`);

    // 确保有足够列
    const requiredCol = 10;
    const curCols = sheet.getMaxColumns();
    if (curCols < requiredCol) {
      const need = requiredCol - curCols;
      sheet.insertColumns(curCols + 1, need);
      SpreadsheetApp.flush();
    }

    // 清除旧内容
    sheet.getRange("F2:G1000").clearContent();
    sheet.getRange("I2:J1000").clearContent();

    // 工具函数：拆分成二维数组 [ [time, title], ... ]
    const splitIntoTimeAndTitle = (arr) => {
      return arr.map(line => {
        const match = line.match(/^([\d:.]+)\s+(.*)$/);
        if (match) {
          return [match[1], match[2]];
        } else {
          return ["", line];
        }
      });
    };

    // 拆分
    const mixTimelineSplit = splitIntoTimeAndTitle(mixTimeline);
    const cleanTimelineSplit = splitIntoTimeAndTitle(cleanTimeline);

    // 强制列为文本格式（防止 "0:03" 被识别为时间戳）
    sheet.getRange(2, 6, mixTimelineSplit.length, 1).setNumberFormat("@");
    sheet.getRange(2, 9, cleanTimelineSplit.length, 1).setNumberFormat("@");

    // 写入 F-G (6–7), I-J (9–10)
    sheet.getRange(2, 6, mixTimelineSplit.length, 2).setValues(mixTimelineSplit);
    sheet.getRange(2, 9, cleanTimelineSplit.length, 2).setValues(cleanTimelineSplit);

    // 表头
    sheet.getRange(1, 6).setValue("🎧 Time");
    sheet.getRange(1, 7).setValue("🎧 FFmpeg Timecode with Needle");
    sheet.getRange(1, 9).setValue("📝 Time");
    sheet.getRange(1,10).setValue("📝 YouTube Song List (No Needle)");

    // 自动调整列宽
    sheet.autoResizeColumns(6, 2); // F-G
    sheet.autoResizeColumns(9, 2); // I-J

    Logger.log("✅ generateMixAndTrackListInSheet 完成");
  } catch (err) {
    Logger.log("❌ generateMixAndTrackListInSheet 出错: " + (err && err.message));
    SpreadsheetApp.getUi().alert("生成时间码出错: " + (err && err.message));
    throw err;
  }
}

//** 插入一个简单 chart 作为画布占位（不是生成图片） */
function insertCanvasHeader(sheet) {
  try {
    const chart = sheet.newChart()
      .setChartType(Charts.ChartType.BAR)
      .addRange(sheet.getRange("A1:B1"))
      .setOption("title","🎨 Mixtape Aesthetic Canvas Placeholder")
      .setOption("width",600)
      .setOption("height",60)
      .build();
    sheet.insertChart(chart);
    Logger.log("插入画布占位 chart 完成");
  } catch (e) {
    Logger.log("插入画布失败: " + e.message);
  }
}
